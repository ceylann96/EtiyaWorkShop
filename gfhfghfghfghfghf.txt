function asal(...numbers) {
numbers.forEach(number => {
  let asalMi=true;
  for(let i=2; i<number; i++) {
    if(number % i == 0){
      asalMi=false;
      break;
    }
  }
  if(asalMi){
    console.log(number+ " : asaldÄ±r");
  }
  else{
    console.log(number+ " :  asal degildir")
  }
})  
}
